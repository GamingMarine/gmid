ACTOR ZombieScientist
{
  obituary "$OB_SCIENTIST"
  health 20
  mass 90
  speed 4
  Radius 20
  Height 52
  painchance 200
  PainChance "Foot", 256
  seesound "scientist/sight"
  painsound "scientist/sight"
  deathsound "scientist/death"
  activesound "scientist/sight"
  Tag "$TAG_SCIENTIST"
  MONSTER
  States
  {
   Spawn:
    SCZA A 10 A_Look
    loop
  See:
    SCZA AAAABBBB 1 A_Chase
	SCZA C 0 A_PlaySound("Cyber/Hoof", CHAN_BODY)
 	SCZA CCCCDDDD 1 A_Chase
	SCZA A 0 A_PlaySound("Cyber/Hoof", CHAN_BODY)
    loop
  Melee:
	SCZA E 0 A_PlaySound("scientist/attack", 2)
    SCZA E 8 A_FaceTarget
    SCZA F 16 A_CustomMeleeAttack (random (2, 6) *5, "scientist/hit", "scientist/whoosh")
    SCZA E 4
    goto See
  Pain:
    SCZA G 3
    SCZA G 3 A_Pain
    goto See
  Pain.Foot:
    SCZA G 2 A_FaceTarget
    SCZA A 0 A_ChangeFlag("NOPAIN",1)
    SCZA A 0 A_Recoil(14)
    SCZA G 33 A_Pain
	SCZA G 0 A_ChangeFlag("NOPAIN",0)
    Goto See
  Death:
	TNT1 A 0 A_Jump(256, "Death1", "Death2")
  Death1:
    SCZA H 5
    SCZA I 5 A_Scream
    SCZA J 5 A_NoBlocking
    SCZA K 5
	SCZA L 5 A_PlaySound("body/hitfloor")
	SCZA M 5
    SCZA N -1
    stop
  Death2:
    FSMS H 5
    FSMS I 5 A_Scream
    FSMS J 5 A_NoBlocking
    FSMS K 5
	FSMS L 0 A_PlaySound("body/hitfloor")
    FSMS L -1
    stop
  Death.Basic:
	TNT1 A 0 A_Jump(256, "Death.Basic1", "Death.Basic2")
  Death.Basic1:
    SCZA H 5
    SCZA I 5 A_Scream
    SCZA J 5 A_NoBlocking
    SCZA K 5
	SCZA L 5 A_PlaySound("body/hitfloor")
	SCZA M 5
	SCZA N 10
	SCZA N 0 A_Jump(16, "KillTaunt1")
    SCZA N -1
    stop
  Death.Basic2:
    FSMS H 5
    FSMS I 5 A_Scream
    FSMS J 5 A_NoBlocking
    FSMS K 5
	FSMS L 10 A_PlaySound("body/hitfloor")
	FSMS L 0 A_Jump(16, "KillTaunt2")
    FSMS L -1
    stop
  KillTaunt1:
	TNT1 A 0 A_PlaySound("player/killtaunt", CHAN_AUTO)
    SCZA N -1
    stop
  KillTaunt2:
	TNT1 A 0 A_PlaySound("player/killtaunt", CHAN_AUTO)
    FSMS L -1
    stop
  Death.Foot:
	TNT1 A 0 A_Jump(256, "Death.Foot1", "Death.Foot2")
  Death.Foot1:
    SCZA H 5 ThrustThingZ(0,16,0,0)
    SCZA I 5 A_Scream
    SCZA J 5 A_NoBlocking
    SCZA K 5
	SCZA L 5 A_PlaySound("body/hitfloor")
	SCZA M 5
	SCZA N 10
	SCZA N 0 A_Jump(16, "KickTaunt1")
    SCZA N -1
    stop
  Death.Foot2:
    FSMS H 5 ThrustThingZ(0,16,0,0)
    FSMS I 5 A_Scream
    FSMS J 5 A_NoBlocking
    FSMS K 5
	FSMS L 10 A_PlaySound("body/hitfloor")
	FSMS L 0 A_Jump(16, "KickTaunt2")
    FSMS L -1
    stop
  KickTaunt1:
	TNT1 A 0 A_PlaySound("player/kicktaunt", CHAN_AUTO)
    SCZA N -1
    stop
  KickTaunt2:
	TNT1 A 0 A_PlaySound("player/kicktaunt", CHAN_AUTO)
    FSMS L -1
    stop
  XDeath:
	TNT1 A 0 A_XScream
	TNT1 A 0 A_NoBlocking
	TNT1 A 0 A_SpawnItem("GMIDGibSpawner")
	TNT1 A 0 A_Jump(16, "GibTaunt")
	Stop
  GibTaunt:
	TNT1 A 40
	TNT1 A 0 A_PlaySound("player/gibtaunt", CHAN_AUTO)
	Stop
  Raise:
    SCZA MLKJIH 5
    goto See
  Death.Fire:
	ZSC1 G 0 A_NoBlocking
	ZSC1 G 0 A_SpawnItem("BurningBoi")
	TNT1 A -1
	Stop
  }
}
