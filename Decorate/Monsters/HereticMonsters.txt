Actor NewHereticImp : HereticImp replaces HereticImp
{
  States
  {
  Death:
	TNT1 A 0 A_GiveToTarget("ScoreItem",200)
    IMPX G 4 A_ImpDeath
    IMPX H 5
    Wait
  XDeath:
	TNT1 A 0 A_GiveToTarget("ScoreItem",200)
    IMPX S 5 A_ImpXDeath1
	IMPX T 0 A_SpawnItem("GMIDGibSpawner_NoSkull")
    IMPX T 5 A_XScream
	IMPX U 5
    IMPX V 5 A_Gravity
    IMPX W 5 
    Wait
  Crash:
	IMPX I 0 A_SpawnItem("GMIDHereticGibSpawner")
    IMPX I 7 A_ImpExplode
    IMPX J 7 A_Scream
    IMPX K 7
	IMPX L 10
	IMPX L 0 A_GiveToTarget("PlayerKillTaunt",1)
    IMPX L -1
    Stop
  XCrash:
    IMPX X 7
    IMPX Y 7
	IMPX Z 1
	IMPX Z 0 A_GiveToTarget("PlayerGibTaunt",1)
    IMPX Z -1
    Stop
  Death.CookieIce:
	"####" "#" 5 A_GenericFreezeDeath
	"----" A 1 A_FreezeDeathChunks
	Wait
  Death.Fire:
  Death.CookieFire:
	IBRN A 0 A_GiveToTarget("ScoreItem",200)
  BurnLoop:
	IBRN A 0 A_PlaySound("himp/burn",CHAN_VOICE)
	IBRN ABCB 5 Bright
	Loop
  Crash.Fire:
  Crash.CookieFire:
	IBRN D 3 Bright
    IBRN E 3 Bright A_Scream
    IBRN FGH 3 Bright
    IBRN I -1
    Stop
  }
}
