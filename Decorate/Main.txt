Actor GMIDFist : Fist Replaces Fist
{
	Weapon.BobRangeX 0.3
	Weapon.BobRangeY 0.5
	Weapon.BobStyle InverseSmooth
	Weapon.BobSpeed 2.0
	Tag "Bare Hands"
	Inventory.PickupMessage "Thou shalt not see this."
	Obituary "%o lost a fistfight with %k."
    +WEAPON.NOALERT
	Weapon.UpSound "knuckled"
	states
	{
		Ready:
			FIST A 1 A_WeaponReady
			TNT1 A 0 A_JumpIfInventory("BotAttack", 1, "Fire")
			loop
		Fire:
			FISU DCBA 1
			TNT1 A 5
			Goto ReadyToGo
		ReadyToGo:
			TNT1 A 0 A_Jump(256, "Fist1", "Fist2")
			Goto Fist1
		Fist1:
			TNT1 A 0 A_PlaySound("fist/normalthrow")
			FIS1 AC 1
			FIS1 E 1 A_CustomPunch(2,0,0,"FistPuff")
			FIS1 FEDCBA 1
			TNT1 A 8 A_Refire("ReadyToGo")
		Goto StopFiring
		Fist2:
			TNT1 A 0 A_PlaySound("fist/normalthrow")
			FIS2 AC 1
			FIS2 E 1 A_CustomPunch(2,0,0,"FistPuff")
			FIS2 FEDCBA 1
			TNT1 A 8 A_Refire("ReadyToGo")
		Goto StopFiring
		StopFiring:
			FISU ABCD 1
			Goto Ready
		AltFire:
			FISU DCBA 1
			TNT1 A 5
			Goto Kick
		Kick:
			TNT1 A 0 A_PlaySound ("weapons/KICK")
			FOOT AB 1
			FOOT C 1
			FOOT D 1
			FOOT D 1 A_CustomPunch(5,0,0,"KickPuff")
			FOOT D 0 A_Quake(1,2,0,10)
			FOOT EF 1
			FOOT FEDCBA 1
			TNT1 A 8
			TNT1 A 0 A_Refire("Kick")
			goto StopFiring
		Deselect:
			FIST A 1 A_Lower
			FIST A 0 A_Lower
			TNT1 A 0 A_TakeInventory("BotMeleeWeapon")
			Loop
		Select:
			FIST A 1 A_Raise
			FIST A 0 A_Raise
			TNT1 A 0 A_GiveInventory("BotMeleeWeapon")
			Loop
		}
}
